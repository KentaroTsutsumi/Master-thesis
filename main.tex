\input{preamble}
%
% ここで \include を無効化する

% \begin{document} は begin.tex 内に書いてある
\input{begin}

\frontmatter

\input{title}
%
% Abstract を目次に加える
\addcontentsline{toc}{chapter}{Abstract}
% Abstract ページ
\input{abstract}
%
% 目次の出力
\tableofcontents
%q
\mainmatter
% 本文
% TMP_chapter.tex をコピーして各章を書く
% 各章は個別にコンパイルしてその章だけのPDFファイルを作成可能
% main.tex をコンパイルすれば全体のPDFになる。

\input{ch1_intro}
\input{ch2_method}
\input{ch3_result}
\input{ch4_conclusion}
%
\appendix
%
% Appendix は本文と同じフォーマット。TMP_chapter.tex をコピーして使う。
\input{appdx1_supplement}

% Acknowledgements を目次に加える
\addcontentsline{toc}{chapter}{Acknowledgements}
% Acknowledgements
\input{acknowledgement}

% \end{document} とbibファイル、bibスタイルの指定は end.tex 内に書いてある
